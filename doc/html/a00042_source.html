01135 Uint32 InterfaceX::getpixel(SDL_Surface *surface, int x, int y)
01136 {

01138     // Here p is the address to the pixel we want to retrieve
01139     Uint8 *p = (Uint8 *)surface->pixels + y * surface->pitch + x * bpp;
01140 
01141     switch(bpp)
01142     {
01143     case 1:
01144         return *p;
01145         break;
01146 
01147     case 2:
01148         return *(Uint16 *)p;
01149         break;
01150 
01151     case 3:
01152         if(SDL_BYTEORDER == SDL_BIG_ENDIAN)
01153             return p[0] << 16 | p[1] << 8 | p[2];
01154         else
01155             return p[0] | p[1] << 8 | p[2] << 16;
01156         _screen = SDL_SetVideoMode( _SCREEN_WIDTH, _SCREEN_HEIGHT, _SCREEN_BPP, SDL_HWSURFACE |SDL_DOUBLEBUF );
01157 
01158         break;

01160     case 4:
01161         return *(Uint32 *)p;
01162         break;

01166     }
