00043                 Similarity() : id(0), sameas(0) {}
00044                 Similarity(int _id, int _sameas) : id(_id), sameas(_sameas) {}

00049     bool is_root_label(int id) {
00050                 return (labels[id].sameas == id);
00051     }
00052     int root_of(int id) {
00053         while (!is_root_label(id)) {
00054             // link this node to its parent's parent, just to shorten
00055             // the tree.
00056             labels[id].sameas = labels[labels[id].sameas].sameas;
00057 
00058             id = labels[id].sameas;
00059         }
00060         return id;
00061     }
00062     bool is_equivalent(int id, int as) {
00063                 return (root_of(id) == root_of(as));
00064     }
00065     bool merge(int id1, int id2) {
00066         if(!is_equivalent(id1, id2)) {
00067             labels[root_of(id1)].sameas = root_of(id2);
00068             return false;
00069         }
00070         return true;
00071     }
00072     int new_label() {
00073                 // Thanks to Pierre Grinspan for suggesting using capacity()
00074                 // instead of size() here.
00075                 if(highest_label+1 >(unsigned char) labels.capacity())
00076                         labels.reserve(highest_label*2);
00077                 labels.resize(highest_label+1);
00078                 labels[highest_label] = Similarity(highest_label);
00079                 return highest_label++;
00080     }

00109 void
00110 ConnectedComponents::label_image(const Tin *img, Tlabel *labelimg,
00111                                  int width, int height, Comparator SAME,
00112                                  const Boolean K8_CONNECTIVITY)
00113 {
00114     const Tin *row = img;
00115     const Tin *last_row = 0;
00116     struct Label_handler {
00117         Label_handler(const Tin *img, Tlabel *limg) :
00118             piximg(img), labelimg(limg) {}
00119         Tlabel &operator()(const Tin *pixp) { return labelimg[pixp-piximg]; }
00120         const Tin *piximg;
00121         Tlabel *labelimg;
00122     } label(img, labelimg);
00123 
00124     clear();
00125 
00126     label(&row[0]) = new_label();
00127 
00128     // label the first row.
00129     for(int c=1; c<width; ++c) {
00130         if(SAME(row[c], row[c-1]))
00131             label(&row[c]) = label(&row[c-1]);
00132         else
00133             label(&row[c]) = new_label();
00134     }
00135 
00136     // label subsequent rows.
00137     for(int r=1; r<height; ++r)    {
00138         // label the first pixel on this row.
00139         last_row = row;
00140         row = &img[width*r];
00141 
00142         if(SAME(row[0], last_row[0]))
00143             label(&row[0]) = label(&last_row[0]);
00144         else
00145             label(&row[0]) = new_label();
00146 
00147         // label subsequent pixels on this row.
00148         for(int c=1; c<width; ++c)      {
00149             int mylab = -1;
00150 
00151             // inherit label from pixel on the left if we're in the same blob.
00152             if(SAME(row[c],row[c-1]))
00153                 mylab = label(&row[c-1]);
00154             for(int d=(K8_CONNECTIVITY?-1:0); d<1; ++d) {
00155                 // if we're in the same blob, inherit value from above pixel.
00156                 // if we've already been assigned, merge its label with ours.
00157                 if(SAME(row[c], last_row[c+d])) {
00158                     if(mylab>=0) merge(mylab, label(&last_row[c+d]));
00159                     else mylab = label(&last_row[c+d]);
00160                 }
00161             }
00162             if(mylab>=0) label(&row[c]) = static_cast<Tlabel>(mylab);
00163             else label(&row[c]) = new_label();
00164 
00165             if(K8_CONNECTIVITY && SAME(row[c-1], last_row[c]))
00166                 merge(label(&row[c-1]), label(&last_row[c]));
00167         }
00168     }
00169 }
